docker run -d -p 9000:9000 --name sonarqube \
  -v sonarqube-keystore:/opt/sonarqube/conf/ssl \
  -e SONAR_WEB_HTTPS_ENABLE=true \
  -e SONAR_WEB_HTTPS_PORT=9000 \
  -e SONAR_WEB_HTTPS_KEYSTOREPASS=<your-keystore-passphrase> \
  -e SONAR_WEB_HTTPS_KEYSTOREFILE=/opt/sonarqube/conf/ssl/keystore.jks \
  sonarqube:latest
